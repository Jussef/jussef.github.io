  try {
    const res = await fetch(url);

  } catch (e) {
    console.error(e); return null;
  }
};

JS LocalStorage Helper 
const store = {
  set: (k, v) =>
    localStorage.setItem(k, JSON.stringify(v)),

  get: (k, fb = null) =>
    JSON.parse(localStorage.getItem(k)) ?? fb,

  del: (k) =>
    localStorage.removeItem(k),
};
